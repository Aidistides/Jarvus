```python
"""JSON-file-based memory store for Root."""
import json
from datetime import datetime, timezone
from pathlib import Path
import uuid

from src import config as cfg


def _load_json(path: Path) -> list[dict]:
    try:
        return json.loads(path.read_text())
    except (FileNotFoundError, json.JSONDecodeError):
        return []


def _save_json(path: Path, data: list[dict]):
    path.write_text(json.dumps(data, indent=2, default=str))


def _now() -> str:
    return datetime.now(timezone.utc).isoformat()


# --- Facts ---

def load_facts() -> list[dict]:
    return _load_json(cfg.FACTS_FILE)


def append_fact(fact: str, source: str) -> dict:
    facts = load_facts()
    entry = {
        "id": f"f_{uuid.uuid4().hex[:6]}",
        "fact": fact,
        "source": source,
        "timestamp": _now(),
    }
    facts.append(entry)
    _save_json(cfg.FACTS_FILE, facts)
    return entry


# --- Decisions ---

def load_decisions() -> list[dict]:
    return _load_json(cfg.DECISIONS_FILE)


def append_decision(decision: str, made_by: str, context: str, follow_up_date: str | None = None) -> dict:
    decisions = load_decisions()
    entry = {
        "id": f"d_{uuid.uuid4().hex[:6]}",
        "decision": decision,
        "made_by": made_by,
        "context": context,
        "timestamp": _now(),
        "status": "active",
        "follow_up_date": follow_up_date,
    }
    decisions.append(entry)
    _save_json(cfg.DECISIONS_FILE, decisions)
    return entry


# --- Learnings ---

def load_learnings() -> list[dict]:
    return _load_json(cfg.LEARNINGS_FILE)


def append_learning(content: str, source: str, confidence: float = 0.8) -> dict:
    learnings = load_learnings()
    entry = {
        "id": f"l_{uuid.uuid4().hex[:6]}",
        "content": content,
        "source": source,
        "timestamp": _now(),
        "confidence": confidence,
    }
    learnings.append(entry)
    _save_json(cfg.LEARNINGS_FILE, learnings)
    return entry


# --- Transcripts ---

def save_transcript(user_id: str, role: str, content: str):
    """Append a message to the user's transcript file for today."""
    user_dir = cfg.TRANSCRIPTS_DIR / str(user_id)
    user_dir.mkdir(parents=True, exist_ok=True)

    today = datetime.now(timezone.utc).strftime("%Y-%m-%d")
    transcript_file = user_dir / f"{today}.json"

    messages = _load_json(transcript_file)
    messages.append({
        "role": role,
        "content": content,
        "timestamp": _now(),
    })
    _save_json(transcript_file, messages)


def load_recent_transcript(user_id: str, limit: int = 20) -> list[dict]:
    """Load the most recent messages for a user across transcript files."""
    user_dir = cfg.TRANSCRIPTS_DIR / str(user_id)
    if not user_dir.exists():
        return []

    all_messages = []
    # Sort files by name (date) descending, load until we have enough
    files = sorted(user_dir.glob("*.json"), reverse=True)
    for f in files:
        messages = _load_json(f)
        all_messages = messages + all_messages
        if len(all_messages) >= limit:
            break

    return all_messages[-limit:]
```

```python
"""ChromaDB-backed knowledge store with OpenAI embeddings."""
import json
from datetime import datetime, timezone

import chromadb
from openai import OpenAI

from src import config as cfg


def _embed(texts: list[str]) -> list[list[float]]:
    """Embed a list of texts using OpenAI embeddings API."""
    client = OpenAI(api_key=cfg.OPENAI_API_KEY)
    response = client.embeddings.create(model=cfg.EMBEDDING_MODEL, input=texts)
    return [item.embedding for item in response.data]


class KnowledgeStore:
    def __init__(self):
        cfg.CHROMA_DIR.mkdir(parents=True, exist_ok=True)
        self._client = chromadb.PersistentClient(path=str(cfg.CHROMA_DIR))
        self._collection = self._client.get_or_create_collection(
            name="enotrium_knowledge",
            metadata={"hnsw:space": "cosine"},
        )

    def add_document(self, source_file: str, chunks: list[str], metadata: dict):
        """Embed and store document chunks, update registry."""
        if not chunks:
            return

        embeddings = _embed(chunks)
        ids = [f"{source_file}_{i}" for i in range(len(chunks))]
        metadatas = [
            {
                "source_file": source_file,
                "source_type": metadata.get("source_type", "text"),
                "uploaded_by": metadata.get("uploaded_by", "unknown"),
                "uploaded_at": datetime.now(timezone.utc).isoformat(),
                "chunk_index": i,
            }
            for i in range(len(chunks))
        ]

        self._collection.upsert(
            ids=ids,
            embeddings=embeddings,
            documents=chunks,
            metadatas=metadatas,
        )

        self._update_registry(source_file, metadata, len(chunks))

    def query(self, text: str, top_k: int = 10) -> list[dict]:
        """Query the knowledge store and return relevant chunks."""
        if self._collection.count() == 0:
            return []

        embedding = _embed([text])[0]
        results = self._collection.query(
            query_embeddings=[embedding],
            n_results=min(top_k, self._collection.count()),
        )

        output = []
        for i in range(len(results["ids"][0])):
            output.append({
                "id": results["ids"][0][i],
                "text": results["documents"][0][i],
                "metadata": results["metadatas"][0][i],
                "distance": results["distances"][0][i] if results.get("distances") else None,
            })
        return output

    def source_count(self) -> int:
        """Return the number of sources in the registry."""
        registry = json.loads(cfg.REGISTRY_FILE.read_text())
        return len(registry)

    def _update_registry(self, source_file: str, metadata: dict, chunk_count: int):
        registry = json.loads(cfg.REGISTRY_FILE.read_text())
        registry.append({
            "source_file": source_file,
            "source_type": metadata.get("source_type", "text"),
            "uploaded_by": metadata.get("uploaded_by", "unknown"),
            "uploaded_at": datetime.now(timezone.utc).isoformat(),
            "chunk_count": chunk_count,
        })
        cfg.REGISTRY_FILE.write_text(json.dumps(registry, indent=2))
```

```python
"""Document ingestion pipeline — extract text, chunk, embed, store."""
from pathlib import Path

import pdfplumber
import pandas as pd

from src.knowledge import KnowledgeStore


def extract_text(file_path: Path) -> str:
    """Extract text from a file based on its extension."""
    suffix = file_path.suffix.lower()

    if suffix == ".pdf":
        return _extract_pdf(file_path)
    elif suffix in (".xlsx", ".xls", ".xlt"):
        return _extract_spreadsheet(file_path)
    elif suffix == ".csv":
        return _extract_csv(file_path)
    else:
        # txt, code, md, etc — read as text
        try:
            return file_path.read_text(errors="replace")
        except Exception:
            return ""


def _extract_pdf(path: Path) -> str:
    try:
        pages = []
        with pdfplumber.open(path) as pdf:
            for page in pdf.pages:
                text = page.extract_text()
                if text:
                    pages.append(text)
        return "\n\n".join(pages)
    except Exception:
        return ""


def _extract_spreadsheet(path: Path) -> str:
    try:
        xl = pd.ExcelFile(path)
        parts = []
        for sheet_name in xl.sheet_names:
            df = xl.parse(sheet_name)
            parts.append(f"## Sheet: {sheet_name}\n{df.to_string()}")
        return "\n\n".join(parts)
    except Exception:
        return ""


def _extract_csv(path: Path) -> str:
    try:
        df = pd.read_csv(path)
        return df.to_string()
    except Exception:
        return ""


def chunk_text(text: str, chunk_size: int = 500, overlap: int = 50) -> list[str]:
    """Split text into overlapping chunks of approximately chunk_size tokens.

    Uses character-based splitting (approx 4 chars per token).
    """
    char_size = chunk_size * 4
    char_overlap = overlap * 4

    if len(text) <= char_size:
        return [text] if text.strip() else []

    chunks = []
    start = 0
    while start < len(text):
        end = start + char_size
        chunk = text[start:end].strip()
        if chunk:
            chunks.append(chunk)
        start = end - char_overlap

    return chunks


def ingest_document(file_path: Path, uploaded_by: str) -> dict:
    """Full ingestion pipeline: extract text, chunk, embed, store in ChromaDB."""
    text = extract_text(file_path)
    if not text.strip():
        return {"source_file": file_path.name, "chunk_count": 0, "error": "No text extracted"}

    # Save processed text
    from src import config as cfg
    processed_file = cfg.PROCESSED_DIR / f"{file_path.stem}.txt"
    processed_file.write_text(text)

    chunks = chunk_text(text)
    if not chunks:
        return {"source_file": file_path.name, "chunk_count": 0, "error": "No chunks generated"}

    store = KnowledgeStore()
    source_type = file_path.suffix.lstrip(".").lower()
    if source_type in ("xls", "xlt"):
        source_type = "xlsx"

    store.add_document(
        source_file=file_path.name,
        chunks=chunks,
        metadata={"source_type": source_type, "uploaded_by": uploaded_by},
    )

    return {
        "source_file": file_path.name,
        "chunk_count": len(chunks),
        "text_length": len(text),
    }
```

```python
"""Post-conversation extraction — pull facts, decisions, learnings from transcripts."""
import json
import logging

from openai import OpenAI

from src import config as cfg
from src.memory import append_fact, append_decision, append_learning

logger = logging.getLogger(__name__)

EXTRACTION_PROMPT = """Analyze this conversation and extract any new information. Return a JSON object with three arrays:

{
  "facts": [{"fact": "...", "source": "conversation with {user}"}],
  "decisions": [{"decision": "...", "made_by": "{user}", "context": "..."}],
  "learnings": [{"content": "...", "confidence": 0.0-1.0}]
}

Rules:
- Only extract genuinely new, specific information — not generic statements
- Facts are concrete data points (numbers, dates, names, statuses)
- Decisions are choices the team has made about what to do
- Learnings are insights that would inform future responses
- If nothing worth extracting, return empty arrays
- Return ONLY the JSON, no other text"""


def _parse_extraction_response(raw: str) -> dict:
    """Parse the LLM extraction response into structured data."""
    try:
        # Handle potential markdown code blocks
        text = raw.strip()
        if text.startswith("```"):
            text = text.split("\n", 1)[1] if "\n" in text else text[3:]
            if text.endswith("```"):
                text = text[:-3]
            text = text.strip()
            if text.startswith("json"):
                text = text[4:].strip()

        data = json.loads(text)
        return {
            "facts": data.get("facts", []),
            "decisions": data.get("decisions", []),
            "learnings": data.get("learnings", []),
        }
    except (json.JSONDecodeError, KeyError, TypeError):
        return {"facts": [], "decisions": [], "learnings": []}


async def extract_from_conversation(messages: list[dict], user_name: str):
    """Run extraction on a conversation and save results to memory."""
    if len(messages) < 2:
        return

    conversation_text = "\n".join(
        f"{m['role'].upper()}: {m['content']}" for m in messages
    )

    prompt = EXTRACTION_PROMPT.replace("{user}", user_name)

    try:
        client = OpenAI(api_key=cfg.OPENAI_API_KEY)
        response = client.chat.completions.create(
            model=cfg.CHAT_MODEL,
            messages=[
                {"role": "system", "content": prompt},
                {"role": "user", "content": conversation_text},
            ],
            temperature=0.1,
            max_tokens=1000,
        )
        raw = response.choices[0].message.content
        extracted = _parse_extraction_response(raw)

        for fact in extracted["facts"]:
            append_fact(fact["fact"], source=fact.get("source", f"conversation with {user_name}"))

        for decision in extracted["decisions"]:
            append_decision(
                decision=decision["decision"],
                made_by=decision.get("made_by", user_name),
                context=decision.get("context", ""),
            )

        for learning in extracted["learnings"]:
            append_learning(
                content=learning["content"],
                source=f"conversation with {user_name}",
                confidence=learning.get("confidence", 0.8),
            )

        logger.info(
            f"Extracted {len(extracted['facts'])} facts, "
            f"{len(extracted['decisions'])} decisions, "
            f"{len(extracted['learnings'])} learnings from conversation with {user_name}"
        )
    except Exception as e:
        logger.error(f"Extraction failed: {e}")
```

```python
"""Agent Core — prompt construction and OpenAI chat completion."""
import logging

from openai import OpenAI

from src import config as cfg
from src.memory import load_facts, load_decisions, load_learnings, load_recent_transcript
from src.knowledge import KnowledgeStore

logger = logging.getLogger(__name__)


def _load_system_prompt() -> str:
    try:
        return cfg.SYSTEM_PROMPT_FILE.read_text()
    except FileNotFoundError:
        return "You are Root, the intelligence agent for Enotrium Syndicate."


def _format_memory_context() -> str:
    """Build the structured memory section of the prompt."""
    sections = []

    decisions = load_decisions()
    active = [d for d in decisions if d.get("status") == "active"]
    recent = sorted(active, key=lambda d: d.get("timestamp", ""), reverse=True)[:cfg.MAX_DECISIONS_IN_PROMPT]
    if recent:
        lines = [f"- {d['decision']} (by {d['made_by']}, {d['timestamp'][:10]})" for d in recent]
        sections.append("## Recent Decisions\n" + "\n".join(lines))

    learnings = load_learnings()
    top = sorted(learnings, key=lambda l: (-l.get("confidence", 0), l.get("timestamp", "")))[:cfg.MAX_LEARNINGS_IN_PROMPT]
    if top:
        lines = [f"- {l['content']} (confidence: {l['confidence']})" for l in top]
        sections.append("## Key Learnings\n" + "\n".join(lines))

    facts = load_facts()
    recent_facts = sorted(facts, key=lambda f: f.get("timestamp", ""), reverse=True)[:cfg.MAX_FACTS_IN_PROMPT]
    if recent_facts:
        lines = [f"- {f['fact']} (source: {f['source']})" for f in recent_facts]
        sections.append("## Key Facts\n" + "\n".join(lines))

    return "\n\n".join(sections)


def _format_rag_context(user_message: str) -> str:
    """Retrieve relevant documents from the knowledge store."""
    try:
        store = KnowledgeStore()
        if store._collection.count() == 0:
            return ""
        results = store.query(user_message, top_k=cfg.RAG_TOP_K)
        if not results:
            return ""
        chunks = []
        for r in results:
            source = r["metadata"].get("source_file", "unknown")
            chunks.append(f"[Source: {source}]\n{r['text']}")
        return "## Relevant Documents\n\n" + "\n\n---\n\n".join(chunks)
    except Exception as e:
        logger.warning(f"RAG retrieval failed: {e}")
        return ""


def build_messages(user_id: str, user_name: str, user_message: str) -> list[dict]:
    """Build the full message list for the OpenAI API call."""
    # 1. System prompt (identity + company context)
    system_prompt = _load_system_prompt()

    # 2. Structured memory
    memory_context = _format_memory_context()
    if memory_context:
        system_prompt += "\n\n" + memory_context

    # 3. RAG context
    rag_context = _format_rag_context(user_message)
    if rag_context:
        system_prompt += "\n\n" + rag_context

    messages = [{"role": "system", "content": system_prompt}]

    # 4. Conversation history
    history = load_recent_transcript(user_id=user_id, limit=cfg.MAX_HISTORY_MESSAGES)
    for msg in history:
        messages.append({"role": msg["role"], "content": msg["content"]})

    # 5. Current user message
    messages.append({"role": "user", "content": user_message})

    return messages


async def get_response(user_id: str, user_name: str, user_message: str) -> str:
    """Build prompt, call GPT 5.4, return the response text."""
    messages = build_messages(user_id, user_name, user_message)

    try:
        client = OpenAI(api_key=cfg.OPENAI_API_KEY)
        response = client.chat.completions.create(
            model=cfg.CHAT_MODEL,
            messages=messages,
            temperature=0.7,
            max_tokens=2000,
        )
        return response.choices[0].message.content
    except Exception as e:
        logger.error(f"OpenAI API call failed: {e}")
        return f"Sorry, I hit an error: {e}"
```

import logging
from pathlib import Path

def _identify_user(update: Update) -> tuple[str, str] | None:
    """Check if the user is in the allowlist. Returns (user_id, display_name) or None."""
    user = update.effective_user
    if not user:
        return None

    # Check by username
    if user.username and user.username.lower().lstrip("@") in cfg.ALLOWED_USERS:
        return str(user.id), cfg.ALLOWED_USERS[user.username.lower().lstrip("@")]

    # Check by user_id (int key)
    if user.id in cfg.ALLOWED_USERS:
        return str(user.id), cfg.ALLOWED_USERS[user.id]

    # Check contact phone (for Tanay)
    # Telegram doesn't expose phone in updates, so we match by user_id
    # Tanay: add your user_id to ALLOWED_USERS after your first /start

    return None


async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle incoming text messages."""
    identity = _identify_user(update)
    if identity is None:
        user = update.effective_user
        logger.warning(f"Unauthorized user: {user.username} (id: {user.id})")
        await update.message.reply_text(
            f"Hey — I don't recognize you yet. Your Telegram user ID is `{user.id}`. "
            f"Ask Tanay to add you to Root's allowlist.",
            parse_mode="Markdown",
        )
        return

    user_id, user_name = identity
    user_message = update.message.text

    if not user_message:
        return

    logger.info(f"Message from {user_name} ({user_id}): {user_message[:100]}")

    # Save user message to transcript
    save_transcript(user_id=user_id, role="user", content=user_message)

    # Get agent response
    response_text = await get_response(user_id=user_id, user_name=user_name, user_message=user_message)

    # Save assistant response to transcript
    save_transcript(user_id=user_id, role="assistant", content=response_text)

    # Send response (split if too long for Telegram's 4096 char limit)
    if len(response_text) <= 4096:
        await update.message.reply_text(response_text)
    else:
        for i in range(0, len(response_text), 4096):
            await update.message.reply_text(response_text[i:i + 4096])

    # Run extraction in background (non-blocking)
    recent = load_recent_transcript(user_id=user_id, limit=10)
    try:
        await extract_from_conversation(recent, user_name)
    except Exception as e:
        logger.error(f"Extraction error: {e}")


async def handle_document(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle file uploads — ingest into knowledge store."""
    identity = _identify_user(update)
    if identity is None:
        await update.message.reply_text("I don't recognize you. Ask Tanay to add you to Root's allowlist.")
        return

    user_id, user_name = identity
    document = update.message.document

    if not document:
        return

    logger.info(f"File from {user_name}: {document.file_name}")
    await update.message.reply_text(f"Got it — processing `{document.file_name}`...", parse_mode="Markdown")

    # Download file
    file = await context.bot.get_file(document.file_id)
    save_path = cfg.SOURCES_DIR / document.file_name
    await file.download_to_drive(save_path)

    # Ingest
    result = ingest_document(save_path, uploaded_by=user_name)

    if result.get("error"):
        await update.message.reply_text(f"Had trouble with `{document.file_name}`: {result['error']}")
    else:
        await update.message.reply_text(
            f"Ingested `{document.file_name}` — {result['chunk_count']} chunks indexed. "
            f"I can now answer questions about this document.",
            parse_mode="Markdown",
        )


async def handle_start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle /start command — show user their ID for allowlist setup."""
    user = update.effective_user
    identity = _identify_user(update)
    if identity:
        _, name = identity
        await update.message.reply_text(
            f"Hey {name}! I'm Root, Enotrium's intelligence agent. "
            f"Send me a message or upload a document to get started."
        )
    else:
        await update.message.reply_text(
            f"Hey! Your Telegram user ID is `{user.id}`. "
            f"Ask Tanay to add this to Root's allowlist in config.py.",
            parse_mode="Markdown",
        )


def create_bot_app() -> Application:
    """Create and configure the Telegram bot application."""
    app = Application.builder().token(cfg.TELEGRAM_BOT_TOKEN).build()
    app.add_handler(CommandHandler("start", handle_start))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    app.add_handler(MessageHandler(filters.Document.ALL, handle_document))
    return app
